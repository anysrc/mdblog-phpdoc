 24: /**
 25:  * A generic ErrorHandler for the PHP engine.
 26:  *
 27:  * Provides five bit fields that control how errors are handled:
 28:  * - thrownErrors: errors thrown as \ErrorException
 29:  * - loggedErrors: logged errors, when not @-silenced
 30:  * - scopedErrors: errors thrown or logged with their local context
 31:  * - tracedErrors: errors logged with their stack trace, only once for repeated errors
 32:  * - screamedErrors: never @-silenced errors
 33:  *
 34:  * Each error level can be logged by a dedicated PSR-3 logger object.
 35:  * Screaming only applies to logging.
 36:  * Throwing takes precedence over logging.
 37:  * Uncaught exceptions are logged as E_ERROR.
 38:  * E_DEPRECATED and E_USER_DEPRECATED levels never throw.
 39:  * E_RECOVERABLE_ERROR and E_USER_ERROR levels always throw.
 40:  * Non catchable errors that can be detected at shutdown time are logged when the scream bit field allows so.
 41:  * As errors have a performance cost, repeated errors are all logged, so that the developer
 42:  * can see them and weight them as more important to fix than others of the same level.
 43:  *
 44:  * @author Nicolas Grekas <p@tchwork.com>
 45:  */

110:     /**
111:      * Registers the error handler.
112:      *
113:      * @param self|null|int $handler The handler to register, or @deprecated (since 2.6, to be removed in 3.0) bit field of thrown levels
114:      * @param bool          $replace Whether to replace or not any existing handler
115:      *

117:      */
118:     public static function register($handler = null, $replace = true)
119:     {
120:         if (null === self::$reservedMemory) {
121:             self::$reservedMemory = str_repeat('x', 10240);
122:             register_shutdown_function(__CLASS__.'::handleFatalError');
123:         }

139:             $handler = $prev[0];

187:      *

195:     {

199:         foreach ($loggers as $type => $log) {
200:             if (!isset($prev[$type])) {
201:                 throw new \InvalidArgumentException('Unknown error type: '.$type);
202:             }
203:             if (!is_array($log)) {
204:                 $log = array($log);
205:             } elseif (!array_key_exists(0, $log)) {
206:                 throw new \InvalidArgumentException('No logger provided');
207:             }
208:             if (null === $log[0]) {

338:     /**
339:      * Handles errors by filtering then logging them according to the configured bit fields.
340:      *
341:      * @param int    $type    One of the E_* constants
342:      * @param string $file
343:      * @param int    $line
344:      * @param array  $context
345:      *
346:      * @return bool Returns false when no handling happens so that the PHP engine can handle the error itself.
347:      *
348:      * @throws \ErrorException When $this->thrownErrors requests so
349:      *
350:      * @internal
351:      */
352:     public function handleError($type, $message, $file, $line, array $context)
353:     {
354:         $level = error_reporting() | E_RECOVERABLE_ERROR | E_USER_ERROR;
355:         $log = $this->loggedErrors & $type;
356:         $throw = $this->thrownErrors & $type & $level;
357:         $type &= $level | $this->screamedErrors;
358: 
359:         if ($type && ($log || $throw)) {
360:             if (PHP_VERSION_ID < 50400 && isset($context['GLOBALS']) && ($this->scopedErrors & $type)) {
361:                 $e = $context;                  // Whatever the signature of the method,
362:                 unset($e['GLOBALS'], $context); // $context is always a reference in 5.3
363:                 $context = $e;
364:             }
365: 
366:             if ($throw) {
367:                 if (($this->scopedErrors & $type) && class_exists('Symfony\Component\Debug\Exception\ContextErrorException')) {
368:                     // Checking for class existence is a work around for https://bugs.php.net/42098
369:                     $throw = new ContextErrorException($this->levels[$type].': '.$message, 0, $type, $file, $line, $context);
370:                 } else {
371:                     $throw = new \ErrorException($this->levels[$type].': '.$message, 0, $type, $file, $line);
372:                 }
373: 
374:                 if (PHP_VERSION_ID <= 50407 && (PHP_VERSION_ID >= 50400 || PHP_VERSION_ID <= 50317)) {
375:                     // Exceptions thrown from error handlers are sometimes not caught by the exception
376:                     // handler and shutdown handlers are bypassed before 5.4.8/5.3.18.
377:                     // We temporarily re-enable display_errors to prevent any blank page related to this bug.
378: 
379:                     $throw->errorHandlerCanary = new ErrorHandlerCanary();
380:                 }
381: 
382:                 throw $throw;
383:             }
384: 
385:             // For duplicated errors, log the trace only once
386:             $e = md5("{$type}/{$line}/{$file}\x00{$message}", true);
387:             $trace = true;
388: 
389:             if (!($this->tracedErrors & $type) || isset($this->loggedTraces[$e])) {
390:                 $trace = false;
391:             } else {
392:                 $this->loggedTraces[$e] = 1;
393:             }
394: 
395:             $e = compact('type', 'file', 'line', 'level');
396: 
397:             if ($type & $level) {
398:                 if ($this->scopedErrors & $type) {
399:                     $e['context'] = $context;
400:                     if ($trace) {
401:                         $e['stack'] = debug_backtrace(true); // Provide object
402:                     }
403:                 } elseif ($trace) {
404:                     $e['stack'] = debug_backtrace(PHP_VERSION_ID >= 50306 ? DEBUG_BACKTRACE_IGNORE_ARGS : false);
405:                 }
406:             }
407: 
408:             if ($this->isRecursive) {
409:                 $log = 0;
410:             } elseif (self::$stackedErrorLevels) {
411:                 self::$stackedErrors[] = array($this->loggers[$type], $message, $e);
412:             } else {
413:                 try {
414:                     $this->isRecursive = true;
415:                     $this->loggers[$type][0]->log($this->loggers[$type][1], $message, $e);
416:                     $this->isRecursive = false;
417:                 } catch (\Exception $e) {
418:                     $this->isRecursive = false;
419: 
420:                     throw $e;
421:                 }
422:             }
423:         }
424: 
425:         return $type && $log;
426:     }

428:     /**
429:      * Handles an exception by logging then forwarding it to an other handler.
430:      *
431:      * @param \Exception $exception An exception to handle
432:      * @param array      $error     An array as returned by error_get_last()
433:      *
434:      * @internal
435:      */
436:     public function handleException(\Exception $exception, array $error = null)
437:     {
438:         $level = error_reporting();
439:         if ($this->loggedErrors & E_ERROR & ($level | $this->screamedErrors)) {
440:             $e = array(
441:                 'type' => E_ERROR,
442:                 'file' => $exception->getFile(),
443:                 'line' => $exception->getLine(),
444:                 'level' => $level,
445:                 'stack' => $exception->getTrace(),
446:             );
447:             if ($exception instanceof FatalErrorException) {
448:                 $message = 'Fatal '.$exception->getMessage();
449:             } elseif ($exception instanceof \ErrorException) {
450:                 $message = 'Uncaught '.$exception->getMessage();
451:                 if ($exception instanceof ContextErrorException) {
452:                     $e['context'] = $exception->getContext();
453:                 }
454:             } else {
455:                 $message = 'Uncaught Exception: '.$exception->getMessage();
456:             }
457:             if ($this->loggedErrors & $e['type']) {
458:                 $this->loggers[$e['type']][0]->log($this->loggers[$e['type']][1], $message, $e);
459:             }
460:         }
461:         if ($exception instanceof FatalErrorException && !$exception instanceof OutOfMemoryException && $error) {
462:             foreach ($this->getFatalErrorHandlers() as $handler) {
463:                 if ($e = $handler->handleError($error, $exception)) {
464:                     $exception = $e;
465:                     break;
466:                 }
467:             }
468:         }
469:         if (empty($this->exceptionHandler)) {
470:             throw $exception; // Give back $exception to the native handler
471:         }
472:         try {
473:             call_user_func($this->exceptionHandler, $exception);
474:         } catch (\Exception $handlerException) {
475:             $this->exceptionHandler = null;
476:             $this->handleException($handlerException);
477:         }
478:     }

480:     /**
481:      * Shutdown registered function for handling PHP fatal errors.
482:      *
483:      * @param array $error An array as returned by error_get_last()
484:      *
485:      * @internal
486:      */
487:     public static function handleFatalError(array $error = null)
488:     {
489:         self::$reservedMemory = '';
490:         $handler = set_error_handler('var_dump', 0);
491:         $handler = is_array($handler) ? $handler[0] : null;
492:         restore_error_handler();
493:         if ($handler instanceof self) {
494:             if (null === $error) {
495:                 $error = error_get_last();
496:             }
497: 
498:             try {
499:                 while (self::$stackedErrorLevels) {
500:                     static::unstackErrors();
501:                 }
502:             } catch (\Exception $exception) {
503:                 // Handled below
504:             }
505: 
506:             if ($error && ($error['type'] & (E_PARSE | E_ERROR | E_CORE_ERROR | E_COMPILE_ERROR))) {
507:                 // Let's not throw anymore but keep logging
508:                 $handler->throwAt(0, true);
509: 
510:                 if (0 === strpos($error['message'], 'Allowed memory') || 0 === strpos($error['message'], 'Out of memory')) {
511:                     $exception = new OutOfMemoryException($handler->levels[$error['type']].': '.$error['message'], 0, $error['type'], $error['file'], $error['line'], 2, false);
512:                 } else {
513:                     $exception = new FatalErrorException($handler->levels[$error['type']].': '.$error['message'], 0, $error['type'], $error['file'], $error['line'], 2, true);
514:                 }
515:             } elseif (!isset($exception)) {
516:                 return;
517:             }
518: 
519:             try {
520:                 $handler->handleException($exception, $error);
521:             } catch (FatalErrorException $e) {
522:                 // Ignore this re-throw
523:             }
524:         }
525:     }

527:     /**
528:      * Configures the error handler for delayed handling.
529:      * Ensures also that non-catchable fatal errors are never silenced.
530:      *
531:      * As shown by http://bugs.php.net/42098 and http://bugs.php.net/60724
532:      * PHP has a compile stage where it behaves unusually. To workaround it,
533:      * we plug an error handler that only stacks errors for later.
534:      *
535:      * The most important feature of this is to prevent
536:      * autoloading until unstackErrors() is called.
537:      */
538:     public static function stackErrors()
539:     {
540:         self::$stackedErrorLevels[] = error_reporting(error_reporting() | E_PARSE | E_ERROR | E_CORE_ERROR | E_COMPILE_ERROR);
541:     }
542: 
543:     /**
544:      * Unstacks stacked errors and forwards to the logger
545:      */
546:     public static function unstackErrors()
547:     {
548:         $level = array_pop(self::$stackedErrorLevels);
549: 
550:         if (null !== $level) {
551:             $e = error_reporting($level);
552:             if ($e !== ($level | E_PARSE | E_ERROR | E_CORE_ERROR | E_COMPILE_ERROR)) {
553:                 // If the user changed the error level, do not overwrite it
554:                 error_reporting($e);
555:             }
