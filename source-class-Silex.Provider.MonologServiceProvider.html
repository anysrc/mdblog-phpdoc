26: class MonologServiceProvider implements ServiceProviderInterface
27: {
28:     public function register(Application $app)
29:     {
30:         $app['logger'] = function () use ($app) {
31:             return $app['monolog'];
32:         };
33: 
34:         if ($bridge = class_exists('Symfony\Bridge\Monolog\Logger')) {
35:             $app['monolog.handler.debug'] = function () use ($app) {
36:                 $level = MonologServiceProvider::translateLevel($app['monolog.level']);
37: 
38:                 return new DebugHandler($level);
39:             };
40:         }
41: 
42:         $app['monolog.logger.class'] = $bridge ? 'Symfony\Bridge\Monolog\Logger' : 'Monolog\Logger';
43: 
44:         $app['monolog'] = $app->share(function ($app) {
45:             $log = new $app['monolog.logger.class']($app['monolog.name']);
46: 
47:             $log->pushHandler($app['monolog.handler']);
48: 
49:             if ($app['debug'] && isset($app['monolog.handler.debug'])) {
50:                 $log->pushHandler($app['monolog.handler.debug']);
51:             }
52: 
53:             return $log;
54:         });
55: 
56:         $app['monolog.handler'] = function () use ($app) {
57:             $level = MonologServiceProvider::translateLevel($app['monolog.level']);
58: 
59:             return new StreamHandler($app['monolog.logfile'], $level, $app['monolog.bubble'], $app['monolog.permission']);
60:         };
61: 
62:         $app['monolog.level'] = function () {
63:             return Logger::DEBUG;
64:         };
65: 
66:         $app['monolog.listener'] = $app->share(function () use ($app) {
67:             return new LogListener($app['logger']);
68:         });
69: 
70:         $app['monolog.name'] = 'myapp';
71:         $app['monolog.bubble'] = true;
72:         $app['monolog.permission'] = null;
73:     }
74: 
75:     public function boot(Application $app)
76:     {
77:         if (isset($app['monolog.listener'])) {
78:             $app['dispatcher']->addSubscriber($app['monolog.listener']);
79:         }
80:     }

82:     public static function translateLevel($name)
83:     {
84:         // level is already translated to logger constant, return as-is
85:         if (is_int($name)) {
86:             return $name;
87:         }
88: 
89:         $levels = Logger::getLevels();
90:         $upper = strtoupper($name);
91: 
92:         if (!isset($levels[$upper])) {
93:             throw new \InvalidArgumentException("Provided logging level '$name' does not exist. Must be a valid monolog logging level.");
94:         }
95: 
96:         return $levels[$upper];
97:     }
