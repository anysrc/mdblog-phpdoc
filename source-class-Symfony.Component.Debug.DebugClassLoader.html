 12: namespace Symfony\Component\Debug;
 13: 
 14: /**
 15:  * Autoloader checking if the class is really defined in the file found.
 16:  *
 17:  * The ClassLoader will wrap all registered autoloaders
 18:  * and will throw an exception if a file is found but does
 19:  * not declare the class.
 20:  *
 21:  * @author Fabien Potencier <fabien@symfony.com>
 22:  * @author Christophe Coevoet <stof@notk.org>
 23:  * @author Nicolas Grekas <p@tchwork.com>
 24:  *
 25:  * @api
 26:  */
 27: class DebugClassLoader
 28: {
 29:     private $classLoader;
 30:     private $isFinder;
 31:     private $wasFinder;
 32:     private static $caseCheck;
 33: 
 34:     /**
 35:      * Constructor.
 36:      *
 37:      * @param callable|object $classLoader
 38:      *
 39:      * @api
 40:      * @deprecated since 2.5, passing an object is deprecated and support for it will be removed in 3.0
 41:      */
 42:     public function __construct($classLoader)
 43:     {
 44:         $this->wasFinder = is_object($classLoader) && method_exists($classLoader, 'findFile');
 45: 
 46:         if ($this->wasFinder) {
 47:             $this->classLoader = array($classLoader, 'loadClass');
 48:             $this->isFinder = true;
 49:         } else {
 50:             $this->classLoader = $classLoader;
 51:             $this->isFinder = is_array($classLoader) && method_exists($classLoader[0], 'findFile');
 52:         }
 53: 
 54:         if (!isset(self::$caseCheck)) {
 55:             self::$caseCheck = false !== stripos(PHP_OS, 'win') ? (false !== stripos(PHP_OS, 'darwin') ? 2 : 1) : 0;
 56:         }
 57:     }
 58: 
 59:     /**
 60:      * Gets the wrapped class loader.
 61:      *
 62:      * @return callable|object a class loader
 63:      *
 64:      * @deprecated since 2.5, returning an object is deprecated and support for it will be removed in 3.0
 65:      */
 66:     public function getClassLoader()
 67:     {
 68:         return $this->wasFinder ? $this->classLoader[0] : $this->classLoader;
 69:     }
 70: 
 71:     /**
 72:      * Wraps all autoloaders
 73:      */
 74:     public static function enable()
 75:     {
 76:         // Ensures we don't hit https://bugs.php.net/42098
 77:         class_exists('Symfony\Component\Debug\ErrorHandler');
 78:         class_exists('Psr\Log\LogLevel');
 79: 
 80:         if (!is_array($functions = spl_autoload_functions())) {
 81:             return;
 82:         }
 83: 
 84:         foreach ($functions as $function) {
 85:             spl_autoload_unregister($function);
 86:         }
 87: 
 88:         foreach ($functions as $function) {
 89:             if (!is_array($function) || !$function[0] instanceof self) {
 90:                 $function = array(new static($function), 'loadClass');
 91:             }
 92: 
 93:             spl_autoload_register($function);
 94:         }
 95:     }
 96: 
 97:     /**
 98:      * Disables the wrapping.
 99:      */
100:     public static function disable()
101:     {
102:         if (!is_array($functions = spl_autoload_functions())) {
103:             return;
104:         }
105: 
106:         foreach ($functions as $function) {
107:             spl_autoload_unregister($function);
108:         }
109: 
110:         foreach ($functions as $function) {
111:             if (is_array($function) && $function[0] instanceof self) {
112:                 $function = $function[0]->getClassLoader();
113:             }
114: 
115:             spl_autoload_register($function);
116:         }
117:     }
118: 
119:     /**
120:      * Finds a file by class name
121:      *
122:      * @param string $class A class name to resolve to file
123:      *
124:      * @return string|null
125:      *
126:      * @deprecated Deprecated since 2.5, to be removed in 3.0.
127:      */
128:     public function findFile($class)
129:     {
130:         if ($this->wasFinder) {
131:             return $this->classLoader[0]->findFile($class);
132:         }
133:     }
134: 
135:     /**
136:      * Loads the given class or interface.
137:      *
138:      * @param string $class The name of the class
139:      *
140:      * @return bool|null True, if loaded
141:      *
142:      * @throws \RuntimeException
143:      */
144:     public function loadClass($class)
145:     {
146:         ErrorHandler::stackErrors();
147: 
148:         try {
149:             if ($this->isFinder) {
150:                 if ($file = $this->classLoader[0]->findFile($class)) {
151:                     require $file;
152:                 }
153:             } else {
154:                 call_user_func($this->classLoader, $class);
155:                 $file = false;
156:             }
157:         } catch (\Exception $e) {
158:             ErrorHandler::unstackErrors();
159: 
160:             throw $e;
161:         }
162: 
163:         ErrorHandler::unstackErrors();
164: 
165:         $exists = class_exists($class, false) || interface_exists($class, false) || (function_exists('trait_exists') && trait_exists($class, false));
166: 
167:         if ('\\' === $class[0]) {
168:             $class = substr($class, 1);
169:         }
170: 
171:         if ($exists) {
172:             $refl = new \ReflectionClass($class);
173:             $name = $refl->getName();
174: 
175:             if ($name !== $class && 0 === strcasecmp($name, $class)) {
176:                 throw new \RuntimeException(sprintf('Case mismatch between loaded and declared class names: %s vs %s', $class, $name));
177:             }
178:         }
179: 
180:         if ($file) {
181:             if (!$exists) {
182:                 if (false !== strpos($class, '/')) {
183:                     throw new \RuntimeException(sprintf('Trying to autoload a class with an invalid name "%s". Be careful that the namespace separator is "\" in PHP, not "/".', $class));
184:                 }
185: 
186:                 throw new \RuntimeException(sprintf('The autoloader expected class "%s" to be defined in file "%s". The file was found but the class was not in it, the class name or namespace probably has a typo.', $class, $file));
187:             }
188:             if (self::$caseCheck && preg_match('#([/\\\\][a-zA-Z_\x7F-\xFF][a-zA-Z0-9_\x7F-\xFF]*)+\.(php|hh)$#D', $file, $tail)) {
189:                 $tail = $tail[0];
190:                 $real = $refl->getFilename();
191: 
192:                 if (2 === self::$caseCheck) {
193:                     // realpath() on MacOSX doesn't normalize the case of characters
194:                     $cwd = getcwd();
195:                     $basename = strrpos($real, '/');
196:                     chdir(substr($real, 0, $basename));
197:                     $basename = substr($real, $basename + 1);
198:                     // glob() patterns are case-sensitive even if the underlying fs is not
199:                     if (!in_array($basename, glob($basename.'*', GLOB_NOSORT), true)) {
200:                         $real = getcwd().'/';
201:                         $h = opendir('.');
202:                         while (false !== $f = readdir($h)) {
203:                             if (0 === strcasecmp($f, $basename)) {
204:                                 $real .= $f;
205:                                 break;
206:                             }
207:                         }
208:                         closedir($h);
209:                     }
210:                     chdir($cwd);
211:                 }
212: 
213:                 if (0 === substr_compare($real, $tail, -strlen($tail), strlen($tail), true)
214:                   && 0 !== substr_compare($real, $tail, -strlen($tail), strlen($tail), false)
215:                 ) {
216:                     throw new \RuntimeException(sprintf('Case mismatch between class and source file names: %s vs %s', $class, $real));
217:                 }
218:             }
219: 
220:             return true;
221:         }
222:     }
223: }
